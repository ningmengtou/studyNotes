##### wapRem js

```js
/*
 *desc:初始化加载
 *@param void;
 *@return void;
 */
$(function() {
  func.changeSize(); //重置rem,1rem=100px
});

/*
 *desc:构造函数 
 *@param void;
 *@return void;
 */
function Func() {};

/*
 *desc:窗口大小改变执行
 *@param void;
 *@return void;
 */
Func.prototype.changeSize = function() {
  func.mobileAnswer();
  $(window).resize(function() {
    func.mobileAnswer();
  });
};

/*
 *desc:重置rem,1rem=100px
 *@param void;
 *@return void;
 */
Func.prototype.mobileAnswer = function() {
  var deviceWidth = document.documentElement.clientWidth;
  if(deviceWidth > 1024) {
    deviceWidth = 1024;
  }
  $("html").css("font-size", deviceWidth / 7.5 + 'px');
};

/*
 *desc:构造函数实例化
 */
var func = new Func();
```
